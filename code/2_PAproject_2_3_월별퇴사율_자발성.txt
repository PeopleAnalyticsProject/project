import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 한글 폰트 설정 (Windows 기준)
plt.rcParams['font.family'] = 'Malgun Gothic'
plt.rcParams['axes.unicode_minus'] = False

# 1. 데이터 로드
file_path = r'C:\Users\user\Desktop\2_PAproject_2_3_EDA.csv'
df = pd.read_csv(file_path)

# 날짜 데이터 형식 변환
df['Hire_Date'] = pd.to_datetime(df['Hire_Date'])
df['Termination_Date'] = pd.to_datetime(df['Termination_Date'])

# 2. 월별 퇴사율(Attrition Rate) 계산
# 분석 기간 설정 (데이터 내 최소 입사일 ~ 최대 퇴사일)
start_period = df['Hire_Date'].min().to_period('M')
end_period = df['Termination_Date'].max().to_period('M')
periods = pd.period_range(start=start_period, end=end_period, freq='M')

attrition_data = []

for period in periods:
    month_start = period.to_timestamp()
    month_end = period.to_timestamp(how='end')

    # 해당 월 시작 시점의 재직자 수 (입사일이 월초 이전이고, 퇴사하지 않았거나 월초 이후에 퇴사한 경우)
    active_at_start = df[
        (df['Hire_Date'] < month_start) &
        ((df['Termination_Date'].isna()) | (df['Termination_Date'] >= month_start))
        ].shape[0]

    # 해당 월 내 퇴사자 수
    terminated_in_month = df[
        (df['Termination_Date'] >= month_start) & (df['Termination_Date'] <= month_end)
        ].shape[0]

    # 퇴사율 계산 (재직자가 0명인 경우 제외)
    rate = (terminated_in_month / active_at_start * 100) if active_at_start > 0 else 0
    attrition_data.append({'Month': str(period), 'Attrition_Rate': rate})

attrition_df = pd.DataFrame(attrition_data)

# 3. 월별 퇴사율 변동 선 그래프 시각화
plt.figure(figsize=(12, 6))
plt.plot(attrition_df['Month'], attrition_df['Attrition_Rate'], marker='o', linestyle='-', color='royalblue',
         linewidth=2)
plt.title('월별 퇴사율 변동 추이 (Monthly Attrition Rate)', fontsize=15, pad=20)
plt.xlabel('연월 (Year-Month)', fontsize=12)
plt.ylabel('퇴사율 (%)', fontsize=12)
plt.xticks(rotation=45)
plt.grid(True, linestyle='--', alpha=0.6)
plt.tight_layout()
plt.show()

# 4. 자발적 vs 비자발적 퇴사 비중 원 그래프 시각화
# 퇴사자(Status가 Voluntary 또는 Involuntary인 경우) 데이터만 추출
termination_types = df[df['Status'].isin(['Voluntary', 'Involuntary'])]['Status'].value_counts()

plt.figure(figsize=(8, 8))
colors = ['#ff9999', '#66b3ff']
plt.pie(termination_types, labels=termination_types.index, autopct='%1.1f%%',
        startangle=140, colors=colors, explode=(0.05, 0), shadow=True)
plt.title('자발적(Voluntary) vs 비자발적(Involuntary) 퇴사 비중', fontsize=15, pad=20)
plt.tight_layout()
plt.show()

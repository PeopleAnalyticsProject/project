import os
import time
import urllib.request
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.common.by import By
from webdriver_manager.chrome import ChromeDriverManager

# 1. 설정값 준비
target_count = 20  # 수집할 이미지 개수
save_path = r"C:\Users\user\Desktop\1"
search_url = "https://search.naver.com/search.naver?sm=tab_hty.top&where=image&ssc=tab.image.all&query=%EB%B3%B4%EB%84%A5%EB%8F%84+%EC%9A%B4%ED%95%99&oquery=%EB%B3%B4%EB%84%A5%EB%8F%84+%ED%83%9C%EC%82%B0&tqi=jjpFqlqV13ZssfuLriK-212632&ackey=ztkheua6"

# 폴더가 없으면 생성
if not os.path.exists(save_path):
    os.makedirs(save_path)

# 2. 브라우저 실행 설정
chrome_options = Options()
# chrome_options.add_argument("--headless") # 창을 띄우지 않고 싶을 때 주석 해제
driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()), options=chrome_options)

try:
    # 3. 페이지 접속
    driver.get(search_url)
    time.sleep(2)  # 이미지 로딩 대기

    # 4. 이미지 요소 찾기
    # 네이버 이미지 검색 결과의 썸네일 클래스 선택자
    img_elements = driver.find_elements(By.CSS_SELECTOR, "._fe_image_tab_content_thumbnail_image")

    print(f"이미지 수집을 시작합니다. (목표: {target_count}장)")

    count = 0
    for i, img in enumerate(img_elements):
        if count >= target_count:
            break

        try:
            # 이미지 원본 주소(src) 추출
            src = img.get_attribute("src")

            # 유효한 URL인지 확인 (base64 데이터 제외)
            if src and "http" in src:
                count += 1
                file_name = f"wh{count}.jpg"
                full_path = os.path.join(save_path, file_name)

                # 파일 저장
                urllib.request.urlretrieve(src, full_path)
                print(f"[{count}/{target_count}] {file_name} 저장 성공")

                # 서버 부하 방지를 위한 미세한 대기
                time.sleep(0.3)

        except Exception as e:
            print(f"{i + 1}번째 이미지 저장 중 오류 발생: {e}")
            continue

    print(f"\n총 {count}장의 사진이 '{save_path}' 폴더에 저장되었습니다.")

finally:
    # 5. 브라우저 종료
    driver.quit()
